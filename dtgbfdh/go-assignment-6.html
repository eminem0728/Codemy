package main import "fmt" type Shape interface { Area() float64 Name() string } type Circle struct { radius float64 } func (c Circle) Area() float64 { return 3.14159 * c.radius * c.radius } func (c Circle) Name() string { return "Circle" } type Rectangle struct { width, height float64 } func (r Rectangle) Area() float64 { return r.width * r.height } func (r Rectangle) Name() string { return "Rectangle" } func printArea(s Shape) { fmt.Printf("%s area: %.2f\n", s.Name(), s.Area()) } func main() { shapes := []Shape{ Circle{radius: 5}, Rectangle{width: 10, height: 5}, } for _, shape := range shapes { printArea(shape) } }
