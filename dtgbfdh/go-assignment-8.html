package main import ( "fmt" "sync" ) func producer(ch chan<- int, wg *sync.WaitGroup) { defer wg.Done() for i := 0; i < 5; i++ { ch <- i fmt.Printf("Produced: %d\n", i) } close(ch) } func consumer(ch <-chan int, wg *sync.WaitGroup) { defer wg.Done() for value := range ch { fmt.Printf("Consumed: %d\n", value) } } func main() { ch := make(chan int, 2) var wg sync.WaitGroup wg.Add(2) go producer(ch, &wg) go consumer(ch, &wg) wg.Wait() fmt.Println("Done") }
